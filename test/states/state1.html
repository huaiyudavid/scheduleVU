   function replaceNameValuePair( href, key, value  )
   {
      var nameValuePair = key + "=" + value;

function updateSelectedTerm()
{
   var select = YAHOO.util.Dom.get("selectedSession");
   var selectedTermCode = YAHOO.mis.student.Util.getSelectedCode("selectedTerm");
   var selectedTermLabel = YAHOO.mis.student.Util.getSelectedValue("selectedTerm");
   var url = "/more/SelectTerm!selectTerm.action?selectedTermCode=" + selectedTermCode;
   var removeOptions = function( select ){
      select.options.length = 0;
   };
   var createOption = function( parent, text, value )
   {
	    var option = new Option();
	    option.value = value;
	    option.text = text;
       parent.options[parent.options.length] = option;
    };
   // on update, empty out and display a loading type message in the select box...
   removeOptions(select);
   createOption( select, "Loading...", "" );
   
   var onSuccess = function( response ){
      var termActivated = YAHOO.lang.JSON.parse(response.responseText);
      
      YAHOO.mis.student.NotificationManager.notify( selectedTermLabel + " term selected" );
      YAHOO.mis.student.Topics.onTermChanged.fire( { term : selectedTermCode, termLabel : selectedTermLabel, active : termActivated } );

      onSessionsRetrieved = function( response )
      {
	      var sessions = YAHOO.lang.JSON.parse(response.responseText);
		  if( sessions != null )
		  {	
			  removeOptions(select);
			  
			  createOption( select, "All Sessions", "" ); 
			  

			  YAHOO.util.Dom.batch( sessions,
				function( session )
			    {
				  createOption( select, session.longDescription, session.code );
			    }  
			  );
		  } 
		  // create an empty option if there are no options/sessions for term
		  if ( select.options.length == 0 ) {
			  createOption( select, "No Sessions Available", "" ); 
		  }
		  YAHOO.mis.student.Topics.onSessionOptionsChanged.fire();
      }; 

	  var updateSessionsUrl = "/more/SelectTerm!updateSessions.action";
      //Update the session list
      YAHOO.mis.student.Ajax.load( updateSessionsUrl , { onSuccess : onSessionsRetrieved } );
    	      
   }

   var onError = function(){
      YAHOO.mis.student.NotificationManager.error( "There was an error setting the selected term, please try again" );
   }
   
   YAHOO.mis.student.Ajax.load( url, { onSuccess : onSuccess, onError : onError } );

}

   YAHOO.util.Event.onContentReady ("selectedTerm", function(){updateSelectedTerm()});
